# DocuSign Configuration for LaborLooker
# Copy these values to your environment variables or .env file

# Required DocuSign Environment Variables
# =====================================

# DocuSign Integration Key (Client ID from your DocuSign app)
DOCUSIGN_CLIENT_ID=your_integration_key_here

# DocuSign Account ID (found in DocuSign admin settings)
DOCUSIGN_ACCOUNT_ID=your_account_id_here

# DocuSign User ID (your user GUID from DocuSign)
DOCUSIGN_USER_ID=your_user_id_here

# DocuSign Base URL (use demo for testing, production for live)
# Demo: https://demo.docusign.net/restapi
# Production: https://na1.docusign.net/restapi (or your region)
DOCUSIGN_BASE_URL=https://demo.docusign.net/restapi

# DocuSign Redirect URI (must match your app settings)
DOCUSIGN_REDIRECT_URI=https://laborlooker.net/docusign/callback

# DocuSign Private Key (RSA private key for JWT authentication)
# This should be the content of your private key file
DOCUSIGN_PRIVATE_KEY=-----BEGIN RSA PRIVATE KEY-----
your_private_key_content_here
-----END RSA PRIVATE KEY-----

# DocuSign Template IDs (create these in DocuSign admin)
# ====================================================

# Contractor Agreement Template
DOCUSIGN_CONTRACTOR_TEMPLATE_ID=your_contractor_template_id

# Client Terms Template  
DOCUSIGN_CLIENT_TEMPLATE_ID=your_client_template_id

# Project Contract Template
DOCUSIGN_PROJECT_TEMPLATE_ID=your_project_template_id

# NDA Agreement Template
DOCUSIGN_NDA_TEMPLATE_ID=your_nda_template_id

# Optional: Client Secret (for authorization code flow)
DOCUSIGN_CLIENT_SECRET=your_client_secret_here

# Configuration Notes:
# ===================
# 1. Replace all "your_*_here" values with actual DocuSign credentials
# 2. Private key should be the full RSA private key including headers
# 3. Template IDs are found in DocuSign Templates section
# 4. For production, change base URL to your region's endpoint
# 5. Redirect URI must be exactly as configured in DocuSign app settings
